    }
}

    \ifthenelse{\equal{#1}{}}{

    }{
        % given

    }
}

% optional first argument for exponent
%    i.e. `\scale{A}{B} = (A/B)` or `\scale[2]{A}{B} = (A/B)^2`
\newcommand{\scale}[3][]{
    \ifthenelse{\equal{#1}{}}{
        % omitted
        \lr{ \frac{#2}{#3} }
    }{
        % given
        {\lr[#1]{ \frac{#2}{#3} }}
    }
}

    }
}
